\vspace{0.2cm}

  \begin{itemize}
    \item \textit{Subnetzmaske} 'spaltet' \textit{IPv4-Adresse} in 

    \begin{itemize}
      \item \footnotesize{\textbf{Netzanteil} (hier `B4.B3.B2`) und}
      \item \textbf{Hostanteil} (hier `B1`)
    \end{itemize}
    \item $\Rightarrow$
    \begin{itemize}
      \item \textbf{Netzadresse} :- Netzanteil $\circ$ kleinste Zahl im Hostanteil 
      \item \textbf{Broadcastadresse} :-  Netzanteil $\circ$  größte Zahl im Hostanteil 
      \item \textbf{Gatewayadresse}  :- `Netzadresse + 1`  | `Broadcastadresse -1` | ...
    \end{itemize}
  \end{itemize}
\end{footnotesize}
