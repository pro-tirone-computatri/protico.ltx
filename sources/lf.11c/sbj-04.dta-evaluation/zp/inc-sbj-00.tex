    \item Maximum [ $\max$ ]

\begin{frame}[fragile]  
  \frametitle{LF11c:04:Datenauswertung:Mathematisch:Beispiel}

  \begin{center}
  \begin{minipage}{0.9\textwidth}
  \begin{minted}[fontsize=\footnotesize,linenos=true]{python}

systole = ( 182, 158, 179, 170 ) # sorted: ( 158, 170, 179, 182 )
range(systole) = 24 # max(systole) - min(systole)
average(systole) = ~172
median(systole) = 174,5
varianz(systole) = 87 
    # ( (158-172)^2 + (170-172)^2 + (179-172)^2 + (182-172)^2 ) / 4
deviance(systole) = 9,32 # sqrt(variance(systole))

pulse = ( 66, 75, 67, 78 ) # sorted ( 66, 67, 75, 78 ) 
range(pulse) = 12 # max(pulse) - min(pulse)
average(pulse) = 71,5
median(pulse) = 71
variance(pulse) = 27 
  # ( (66-72)^2 + (75-72)^2 + (67-72)^2 + (78-72)^2 ) / 4
deviance(pulse) = 5,2 # sqrt(variance(pulse))

covariance(systole,pulse) = -28 # ... see oraltrack.pdf
correlation_coefficient(systole,pulse) = -0.57 
  # covariance(s,p) / (deviance(s) * deviance(p))
  \end{minted}
  \end{minipage}
  \end{center}
